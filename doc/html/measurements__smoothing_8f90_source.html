201  subroutine lowess(x, y, f, nsteps, delta, ys, rw, res)
202  ! Arguments
203  real(real64), intent(in), dimension(:) :: x, y
204  real(real64), intent(in) :: f
205  integer(int32), intent(in) :: nsteps
206  real(real64), intent(in) :: delta
207  real(real64), intent(out), dimension(:) :: ys, rw, res
208 
209  ! Parameters
210  real(real64), parameter :: zero = 0.0d0
211  real(real64), parameter :: one = 1.0d0
212  real(real64), parameter :: three = 3.0d0
213  real(real64), parameter :: p001 = 1.0d-3
214  real(real64), parameter :: p999 = 0.999d0
215 
216  ! Local Variables
217  logical :: ok
218  integer(int32) :: iter, i, j, n, nleft, nright, ns, last, m1, m2
219  real(real64) :: d1, d2, denom, alpha, cut, eps, cmad, c1, c9, r
220 
221  ! Initialization
222  n = size(x)
223  ns = max(min(int(f * real(n, real64), int32), n), 2)
224  eps = epsilon(eps)
225 
226  ! Quick Return
227  if (n < 2) then
228  ys = y
229  return
230  end if
231 
232  ! Process
233  do iter = 1, nsteps + 1
234  nleft = 1
235  nright = ns
236  last = 0
237  i = 1
238  do
239  do while (nright < n)
240  d1 = x(i) - x(nleft)
241  d2 = x(nright+1) - x(i)
242  if (d1 <= d2) exit
243  nleft = nleft + 1
244  nright = nright + 1
245  end do
246 
247  call lowest(x, y, x(i), ys(i), nleft, nright, res, iter > 1, &
248  rw, ok)
249  if (.not.ok) ys(i) = y(i)
250  if (last < i - 1) then
251  denom = x(i) - x(last)
252  do j = last + 1, i - 1
253  alpha = (x(j) - x(last)) / denom
254  ys(j) = alpha * ys(i) + (one - alpha) * ys(last)
255  end do
256  end if
257  last = i
258  cut = x(last) + delta
259  do i = last + 1, n
260  if (x(i) > cut) exit
261  if (abs(x(i) - x(last)) < eps) then
262  ys(i) = ys(last)
263  last = i
264  end if
265  end do
266  i = max(last + 1, i - 1)
267 
268  if (last >= n) exit
269  end do
270 
271  res = y - ys
272  if (iter > nsteps) exit
273  rw = abs(res)
274  call sort(rw, .true.)
275  m1 = 1 + n / 2
276  m2 = n - m1 + 1
277  cmad = three * (rw(m1) + rw(m2))
278  c9 = p999 * cmad
279  c1 = p001 * cmad
280  do i = 1, n
281  r = abs(res(i))
282  if (r <= c1) then
283  rw(i) = one
284  else if (r > c9) then
285  rw(i) = zero
286  else
287  rw(i) = (one - (r / cmad)**2)**2
288  end if
289  end do
290  end do
291  end subroutine

324  subroutine lowest(x, y, xs, ys, nleft, nright, w, userw, rw, ok)
325  ! Arguments
326  real(real64), intent(in), dimension(:) :: x, y, rw ! N ELEMENT
327  real(real64), intent(in) :: xs
328  real(real64), intent(out) :: ys
329  integer(int32), intent(in) :: nleft, nright
330  real(real64), intent(out), dimension(:) :: w ! N ELEMENT
331  logical, intent(in) :: userw
332  logical, intent(out) :: ok
333 
334  ! Parameters
335  real(real64), parameter :: zero = 0.0d0
336  real(real64), parameter :: one = 1.0d0
337  real(real64), parameter :: p001 = 1.0d-3
338  real(real64), parameter :: p999 = 0.999d0
339 
340  ! Local Variables
341  integer(int32) :: j, n, nrt
342  real(real64) :: range, h, h9, h1, a, b, c, r
343 
344  ! Initialization
345  n = size(x)
346  range = x(n) - x(1)
347  h = max(xs - x(nleft), x(nright) - xs)
348  h9 = p999 * h
349  h1 = p001 * h
350  a = zero
351 
352  ! Process
353  do j = nleft, n
354  w(j) = zero
355  r = abs(x(j) - xs)
356  if (r <= h9) then
357  if (r > h1) then
358  w(j) = (one - (r / h)**3)**3
359  else
360  w(j) = one
361  end if
362  if (userw) w(j) = rw(j) * w(j)
363  a = a + w(j)
364  else if (x(j) > xs) then
365  exit
366  end if
367  end do
368 
369  nrt = j - 1
370  if (a <= zero) then
371  ok = .false.
372  else
373  ok = .true.
374  w(nleft:nrt) = w(nleft:nrt) / a
375  if (h > zero) then
376  a = zero
377  do j = nleft, nrt
378  a = a + w(j) * x(j)
379  end do
380  b = xs - a
381  c = zero
382  do j = nleft, nrt
383  c = c + w(j) * (x(j) - a)**2
384  end do
385  if (sqrt(c) > p001 * range) then
386  b = b / c
387  do j = nleft, nrt
388  w(j) = w(j) * (one + b * (x(j) - a))
389  end do
390  end if
391  end if
392  ys = zero
393  do j = nleft, nrt
394  ys = ys + w(j) * y(j)
395  end do
396  end if
397  end subroutine

444  ! Index the buffer

446
